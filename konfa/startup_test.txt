# Стартовый скрипт для тестирования команд Этапа 5
# Тестирование команды mkdir

# === ТЕСТ 1: Создание простой директории ===
echo "Тест 1: Создание директории в текущем каталоге"
pwd
mkdir test_dir1
ls -l

# === ТЕСТ 2: Создание вложенных директорий ===
echo "\nТест 2: Создание вложенных директорий"
cd test_dir1
mkdir subdir1
mkdir subdir2
ls -l
cd ..

# === ТЕСТ 3: Создание с абсолютным путем ===
echo "\nТест 3: Создание с абсолютным путем"
mkdir /temp_test
cd /temp_test
pwd
ls -l
cd /

# === ТЕСТ 4: Обработка ошибок ===
echo "\nТест 4: Проверка обработки ошибок"

# Попытка создать существующую директорию
echo "Попытка создать существующую директорию:"
mkdir test_dir1

# Создание в несуществующем пути
echo "Создание в несуществующем пути:"
mkdir invalid/path

# Некорректное имя
echo "Создание с некорректным именем:"
mkdir "dir/with/slash"

# === ТЕСТ 5: Комбинирование с другими командами ===
echo "\nТест 5: Комбинирование команд"
cd /home/user1
mkdir new_project
cd new_project
mkdir src
mkdir docs
mkdir tests
touch main.py
ls -l

# === ТЕСТ 6: История команд ===
echo "\nТест 6: Проверка истории команд"
history

# === ИТОГ ===
echo "\n=== ИТОГОВЫЙ ОБЗОР СИСТЕМЫ ==="
pwd
conf-dump
echo "Все директории созданы только в памяти!"
